## <a name="example"></a><span data-ttu-id="7ba57-107">Exemple</span><span class="sxs-lookup"><span data-stu-id="7ba57-107">Example</span></span>

```csharp  
using System;  
  
[assembly:CLSCompliant(true)]  
public abstract class I  
{  
    [CLSCompliant(false)]  
    public abstract int M();   // CS3011  
  
    // OK  
    [CLSCompliant(false)]  
    public void M2()  
    {  
    }  
}  
  
public class C : I  
{  
    public override int M()  
    {  
        return 1;  
    }  
  
    public static void Main()  
    {  
    }  
}  
```
